}); const labels = Object.keys(expenses).map(cat => CONFIG.CATEGORY_NAMES[cat] || cat); const data = Object.values(expenses); const colors = ['#ef4444', '#f59e0b', '#3b82f6', '#10b981', '#8b5cf6', '#ec4899', '#6366f1', '#14b8a6'];
